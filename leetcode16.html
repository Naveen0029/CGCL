Problem 67

Number of Operations to Make Network Connected

Algorithm

count number of connected component - 1 

                class Solution {
                    public:
                        void dfs(vector<vector<int>> &adj,vector<bool> &visited,int v){
                            visited[v]=true;
                        
                           for(auto itr=adj[v].begin();itr!=adj[v].end();itr++){
                               if(!visited[*itr])
                                dfs(adj,visited,*itr); 
                                 
                           } 
                        
                            return;
                        }
                        int makeConnected(int n, vector<vector<int>>& connections) {
                            if(connections.size()<n-1)
                                return -1;
                            vector<vector<int>> adj(n);
                            for(auto x : connections){
                                adj[x[0]].push_back(x[1]);
                                adj[x[1]].push_back(x[0]);
                            }
                            vector<bool> visited(n,false);
                            
                            int count=0;
                            for(int i=0;i<n;i++){
                                if(!visited[i]){
                                    dfs(adj,visited,i);
                                    count++;
                                }
                                
                            }
                            
                          return count-1;
                           
                        }
                    };

Problem 68

Path Sum

Algorithm

                class Solution {
                    public:
                        bool hasPathSum(TreeNode* root, int sum) {
                          
                            if(root==NULL)
                                return false;
                            
                            
                            sum=sum-(root->val);
                            if(sum==0&&(root->left==NULL&&root->right==NULL))
                                return true; 
                                
                            return hasPathSum(root->left,sum)||hasPathSum(root->right,sum);
                        }
                    };

Algorithm

                class Solution {
                    public:
                        bool hasPathSum(TreeNode* root, int targetSum) {
                            if(root==NULL)return false;
                            if(root->left==NULL&&root->right==NULL){
                                if(root->val==targetSum)return true;
                                else return false;
                            }
                            bool a=hasPathSum(root->left,targetSum-root->val);
                            bool b=hasPathSum(root->right,targetSum-root->val);
                            
                            return a||b;  
                        }
                    };

                class Solution {
                    public:
                        string convert(int n){
                            ostringstream str1;
                            str1<<n;
                            string s=str1.str();
                            
                            return s;
                        }
                        void makeString(TreeNode * root,string fs,vector<string> &v){
                            if(root==NULL)
                                return;
                            if(root->left==NULL&&root->right==NULL){
                                fs=fs+"->"+convert(root->val);
                                v.push_back(fs);
                            }
                            string s=convert(root->val);
                            if(fs=="")
                                fs=s;
                            else fs=fs+"->"+s;
                            
                            makeString(root->left,fs,v);
                            makeString(root->right,fs,v);
                            return;
                        }
                        vector<string> binaryTreePaths(TreeNode* root) {
                            if(root==NULL)
                                return {};
                            if(root->left==NULL&&root->right==NULL)
                                return {convert(root->val)};
                            vector<string> v;
                            makeString(root,"",v);
                            return v;
                        }
                    };

Problem 70

Palindrome Partitioning

Algorithm

Using Backtracking. 

                class Solution {
                    public:
                        bool palindrome(string s){
                            string a=s;
                            reverse(s.begin(),s.end());
                            if(s.compare(a)==0){
                    
                                return true;
                            }
                            else return false;
                        }
                        void check(string s,vector<string> rv,vector<vector<string>> &v){
                            if(s==""){
                                v.push_back(rv);
                                return;
                            }
                            for(int i=1;i<=s.size();i++){
                                  if(palindrome(s.substr(0,i))){
                                    rv.push_back(s.substr(0,i));
                                   check(s.substr(i,s.size()-i),rv,v);
                                   rv.pop_back();
                                   
                                   }
                                
                            }
                          
                            return;
                        }
                        vector<vector<string>> partition(string s) {
                            if(s.size()==1)
                                return {{s}};
                            
                            vector<string> rv;
                            vector<vector<string>> v;
                            check(s,rv,v);
                            return v;
                        }
                    };

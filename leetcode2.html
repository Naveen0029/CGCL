Problem 11

Pascal's Triangle II

Algorithm

use 2d array and to find arr[i][j] we have to add arr[i-1][j-1]+arr[i-1][j]. 

                class Solution {
                    public:
                        vector<int> getRow(int rowIndex) {
                            
                            vector<int> vec;
                            int n=rowIndex;
                            int v[n+1][n+1];
                        
                            for(int i=0;i<=n;i++){
                                for(int j=0;j<=n;j++){
                                    if(j==0)
                                        v[i][0]=1;
                                    else
                                        v[i][j]=0;
                                }
                            }
                            
                            for(int i=1;i<=n;i++){
                                for(int j=1;j<=i;j++){
                                    v[i][j]=v[i-1][j-1]+v[i-1][j];
                                }
                            }
                            
                            for(int j=0;j<=n;j++){
                                vec.push_back(v[n][j]);
                            }
                            
                            return vec;
                            
                        }
                    };

Problem 12

Maximum Depth of Binary Tree

Algorithm

we use dfs(depth first traversal) approch 

                class Solution {
                    public:
                        int maxDepth(TreeNode* root) {
                            if(root==NULL)
                                return 0;
                           
                            
                         int   ldepth=maxDepth(root->left);
                          int  rdepth=maxDepth(root->right);
                            
                             if(ldepth>=rdepth)
                                return ldepth+1;
                            else
                                return rdepth+1;
                            
                            
                        }
                    };

Problem 13

Longest Palindrome

Algorithm

First we count the alphabets using map and the second thing is count the total number of even alphabet and if we get the odd number of alphabet then first we check the previous odd value is less than the current odd value.if the prev odd value is less than the current odd value then we make the previous odd value even and add it to our final answer and initialize the Odd_val to the current odd val.else if the current odd value is not greater than the previous odd value then we simply add the even value of this odd number (odd-1). 

                class Solution {
                    public:
                        int longestPalindrome(string s) {
                            if(s.size()==0)return 0;
                            int count=0,odd_val=INT_MIN;
                            
                            map<char,int> mp;
                            for(int i=0;i<s.size();i++){
                                mp[s[i]]++;
                            }
                            for(auto itr=mp.begin();itr!=mp.end();itr++){
                                if(itr->second%2==0){
                                    count+=itr->second;
                                }
                                else{
                                    if(odd_val<itr->second){
                                        if(odd_val!=INT_MIN)count+=odd_val-1;
                                        
                                         odd_val=itr->second;
                                         }
                                    else
                                        if(itr->second>0)count+=itr->second-1;
                                          
                                }
                            }
                            if(odd_val!=INT_MIN)
                            return count+odd_val;
                            
                            return count;
                        }
                    };

Algorithm

The concept is same.we first always add the even number of odd value. itr.second/2 is odd or even. but by multiplying it with 2 we make even. why we are using itr.second/2 because it gives the same answer for 7 and 6 like number .so what is going here?it means we are considering only even part of the number. so when our answer is even and itr.second is odd then we can increase ans by 1.because particluar number is odd and ans can be divided into two prats. even odd even is palindrome. 

                class Solution {
                    public:
                        int longestPalindrome(string s) {
                            map<char,int> mp;
                            
                            for(int i=0;i<s.length();i++){
                                mp[s[i]]++;
                            }
                            
                            int ans=0;
                            for(auto itr=mp.begin();itr!=mp.end();itr++){
                                ans+=itr->second/2*2;
                                if(ans%2==0&&itr->second%2==1)
                                    ans++;
                            }
                            return ans;
                        }
                    };

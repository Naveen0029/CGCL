                class Solution {
                    public:
                        string removeKdigits(string num, int k) {
                            if(num.size()<=k)return "0";
                           int i=0;
                            stack<char> s;
                            while(i<num.size()){
                                while(!s.empty()&&k>0&&(s.top()-'0')>(num[i]-'0')&&s.top()!='0'){
                                    s.pop();
                                    k--;
                                }
                                
                                s.push(num[i]);
                                i++;
                            }
                            
                            string str="";
                            while(!s.empty()){
                                str=s.top()+str;
                                s.pop();
                            }
                            
                           i=0;
                            //for handling number starting from 0
                           while(i<str.size()){
                                if(str[i]=='0')str.erase(i,1);
                                else break;
                            }
                            
                            if(str.size()<=k)return "0";
                            
                            //for handling increasing order
                            while(k>0){
                                str.pop_back();
                                k--;
                            }
                            return str;
                        }
                    };

                            if(rv.size()==k){
                                v.push_back(rv);
                                return ;
                            }
                            for(int i=index;i<=n;i++){
                                rv.push_back(i);
                                Backtrack(i+1,n,k,rv,v);
                                rv.pop_back();
                            }
                            
                        }
                        vector<vector<int>> combine(int n, int k) {
                            vector<int> rv;
                            vector<vector<int>> v;
                            Backtrack(1,n,k,rv,v);
                            return v;
                        }
                    };

                class Solution {
                    public:
                        vector<vector<int>> subsets(vector<int>& nums) {
                            vector<int> rv;
                            vector<vector<int>> v;
                            int n=(1<<nums.size());
                    
                            for(int i=0;i<n;i++){
                                int k=i;
                                int j=0;
                                while(k){
                                    if(k&1){
                                        rv.push_back(nums[j]);
                                    }
                                    j++;
                                    k=k>>1;
                        
                                }
                                v.push_back(rv);
                                rv.clear();
                            }
                            return v;

take care of leading zeros and range of numbers[0,255] and it also contain four integer

                class Solution {
                    public:
                       
                        int conversion(string s){
                            stringstream str(s);
                            int x=0;
                            str>>x;
                            return x;
                        }
                        void Backtrack(string s,string str,int cnt,vector<string> &v){
                            if(cnt==4&&s.size()>3||s.size()==0)return ;
                            
                            if(cnt==4&&s.size()<=3&&(conversion(s))<=255){
                                if(s.size()>1&&s[0]=='0')return ;
                                
                                v.push_back(str+s);
                                return ;
                            }
                       
                            for(int i=1;i<=(3<s.size()?3:s.size());i++){
                                string temp=s.substr(0,i);
                               if(conversion(temp)>255||(temp.size()>1&&temp[0]=='0'))continue;
                                
                                
                                Backtrack(s.substr(i),str+temp+'.',cnt+1,v);
                                
                            }
                            return ;
                        }
                        vector<string> restoreIpAddresses(string s) {
                            vector<string> v;
                            Backtrack(s,"",1,v);
                            return v;
                        }
                    };

                #define ll int
                class Solution {
                public:
                
                    ll conversion(string s){
                        stringstream str(s);
                       ll x=0;
                        str>>x;
                        return x;
                    }
                    bool Backtrack(string s,vector<int>&v){
                        if(s.empty()&&v.size()>=3)return true;
                        
                        for(ll i=1;i<=s.size();i++){
                            string temp=s.substr(0,i);
                            ll val=conversion(temp);
                            if(val>=INT_MAX)return false;
                            
                            if((temp.size()>1&&temp[0]=='0'))return false;
                            if(v.size()>=2){
                                long long int a=v[v.size()-1];
                                long long int b=v[v.size()-2];
                                long long int res=a+b;
                                if(res>=INT_MAX)return false;
                         
                                if(res<val)return false;
                                if(res!=val)continue;
                                
                            }
                            
                            v.push_back(val);
                            if(Backtrack(s.substr(i),v))return true;
                            
                            v.pop_back();
                
                        }
                        return false;
                    }
                    vector<int> splitIntoFibonacci(string S) {
                        if(S.size()<3)return {};
                        vector<int> v;
                        Backtrack(S,v);
                       
                        
                        return v;
                    }
                };

Single Number

                class Solution {
                    public:
                        int singleNumber(vector<int>& nums) {
                            int ans=nums[0];
                            for(int i=1;i<nums.size();i++){
                                ans=ans^nums[i];
                            }
                            return ans;
                        }
                    };

                class Solution {
                    public:
                        int singleNumber(vector<int>& nums) {
                            map<int,int> mp;
                            int a=0;
                            int b=0;
                            
                            for(int i=0;i<nums.size();i++){
                                if(mp.find(nums[i])==mp.end()){
                                    a+=nums[i];
                                    mp[nums[i]]++;
                                }
                                else b+=nums[i];
                            }
                            return a-b;
                        }
                    };

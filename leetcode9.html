Problem 36

Majority Element

Algorithm

we are assuming that on our first index the majority element is present.so whenever the same element apperas again we increase the count.And if other element elements appears then we decrease the count.But as we know their are n/2+1 majority elements.we will definitely find the majority element. 

                class Solution {
                    public:
                        int majorityElement(vector<int>& nums) {
                            int n=nums.size();
                            if(n<1)
                                return {};
                            
                            int maj_num=nums[0];
                            int count=1;
                            for(int i=1;i<n;i++){
                                if(maj_num==nums[i])
                                    count++;
                                else if(count==0){
                                    maj_num=nums[i];
                                    count=1;
                                }
                                else{
                                    count--;
                                }
                                    
                          }
                           return maj_num; 
                        }
                    };

Algorithm

Time complexity O(nlogn) if we sort the array then at n/2th index our majority element is present. 

                class Solution {
                    public:
                        int majorityElement(vector<int>& nums) {
                            sort(nums.begin(),nums.end());
                            int n=nums.size();
                            return nums[n/2];
                        }
                    };

Problem 37

3Sum Closest

Algorithm

First sort the array and consider the three pointer.First pointer is fixed and second and third pointer always towards each other.and we have to check the minimum absolute difference of the sum of these three pointer with target. abs(target-(nums[first]+nums[second]+nums[third])). 

                class Solution {
                    public:
                        int threeSumClosest(vector<int>& nums, int target) {
                            sort(nums.begin(),nums.end());
                            int n=nums.size();
                            int diff=INT_MAX;
                            int ans;
                            for(int i=0;i<n;i++){
                            int j=i+1;
                            int k=n-1;
                            while(j<k){
                               int  sum=nums[i]+nums[j]+nums[k];
                                if(abs(sum-target)<abs(diff)){
                                    diff=sum-target;
                                    ans=sum;
                                    
                                }
                                if(sum<target)
                                    j++;
                                else{
                                    k--;
                                }
                            }
                            }     
                            return ans;
                        }
                    };

Problem 38

Power of Two

Algorithm

0001-1 0010-2  0100-4  1000-8  so we are seeing that if a number is power of two then the only one bit is set remaining are not set. so when we reduce 1 from n then it all other bit become set except the bit which is set earlier. 0111-7  when n&n-1 is equal to zero it means the number is power of 2.
0111-7  0110-6 
7&6 is not equal to zero 

                class Solution {
                    public:
                        bool isPowerOfTwo(int n) {
                            if(n<=0)
                                return false;
                            if((n&n-1)==0)
                                return true;
                            else 
                                return false;
                        }
                    };

Problem 39

4Sum

Algorithm

Fix the one index and caluclate the three sum. 

                class Solution {
                    public:
                        vector<vector<int>> threeSum(vector<int>& nums,int target,
                        int j,vector<vector<int>> &vec,map<vector<int>,int> &m){
                            vector<int> v;
                            int s=nums.size();
                            for(int i=j+1;i<s;i++){
                                int n=i+1;
                                int k=s-1;
                                
                                while(n<k){
                                   int sum=nums[i]+nums[n]+nums[k];
                                    if(sum<target){
                                        n++;
                                    }
                                    else if(sum>target)
                                        k--;
                                    else{
                                        v.push_back(nums[j]);
                                        v.push_back(nums[i]);
                                        v.push_back(nums[n]);
                                        v.push_back(nums[k]);
                                        if(m.find(v)==m.end()){
                                        vec.push_back(v);
                                         m[v]++;}
                                    
                                        v.clear();
                                        n++;k--;
                                    }
                               }
                            }
                            return vec;
                        }
                        vector<vector<int>> fourSum(vector<int>& nums, int target) {
                            if(nums.size()<4)
                                return {};
                            
                            vector<vector<int>> vec;
                            map<vector<int>,int> mp;
                            sort(nums.begin(),nums.end());
                            for(int i=0;i<nums.size()-3;i++){
                                threeSum(nums,target-nums[i],i,vec,mp);
                            }
                            return vec;
                        }
                    };

No of index fixed is always equal to k-2.so it means we have to find two sum after fixing two index while finding 4Sum. 

 class Solution {
                public:
                    vector<vector<int>> twoSum(vector<int> &nums,int target,int start){
                        vector<vector<int>> vec;
                        vector<int> v;
                        int end=nums.size()-1;
                        int low=start;
                        while(low<end){
                            int sum=nums[low]+nums[end];
                            if(sum<target||(low>start&&nums[low]==nums[low-1]))low++;
                            else if(sum>target||((end<nums.size()-1)&&nums[end]==nums[end+1]))end--;
                            else{
                                
                                v.push_back(nums[low++]);
                                v.push_back(nums[end--]);
                                vec.push_back(v);
                                v.clear();
                            }
                        }
                        return vec;
                    }
                    vector<vector<int>> kSum(vector<int>&nums,int target,int start,int k){
                        vector<vector<int>> res;
                         vector<int> v;
                        if(start==nums.size()||nums[start]*k>target||target>nums[nums.size()-1]*k)
                        return res;
                        if(k==2)return twoSum(nums,target,start);
                        for(int i=start;i<nums.size();i++){
                            if(i==start||(i>0&&nums[i-1]!=nums[i])){
                                for(vector<int> itr:kSum(nums,target-nums[i],i+1,k-1)){
                                   
                                    v.push_back(nums[i]);
                                    int k=itr.size()-1;
                                    reverse(itr.begin(),itr.end());
                                    while(!itr.empty()){
                                        v.push_back(itr[k--]);
                                        itr.pop_back();
                                    }
                                    res.push_back(v);
                                    
                                    v.clear();
                                }
                            }
                        }
                        return res;
                    }
                    vector<vector<int>> fourSum(vector<int>& nums, int target) {
                        sort(nums.begin(),nums.end());
                        return kSum(nums,target,0,4);
                    }
                    
                };

Problem 40

Flatten Binary Tree to Linked List

Algorithm

Using stack 

                class Solution {
                    public:
                        void fill_flatten(TreeNode *root,queue<TreeNode *> &s){
                            if(root==NULL)
                                return ;
                            
                            s.push(root);
                            fill_flatten(root->left,s);
                            fill_flatten(root->right,s);
                            return;
                        }
                        void flatten(TreeNode* root) {
                            
                            queue<TreeNode *> s;
                            fill_flatten(root,s);
                            TreeNode *dummy=new TreeNode(0);
                            while(!s.empty()){
                                TreeNode *n=s.front();
                                s.pop();
                                dummy->right=n;
                                dummy->left=NULL;
                                dummy=dummy->right;
                            }
                        }
                    };

Algorithm

Using Recursion 

 
                class Solution {
                    public:
                        TreeNode *flatt(TreeNode *root){
                             if(root==NULL)return root;
                            
                            TreeNode *a=flatt(root->right);
                            TreeNode *b=flatt(root->left);
                            TreeNode *temp=b;
                            if(b){
                                while(b->right!=NULL)b=b->right;
                                b->right=a;
                            }
                            if(temp)root->right=temp;
                            root->left=NULL;
                            return root;
                        }
                        void flatten(TreeNode* root) {
                            if(root==NULL)return ;
                            flatt(root);
                            return;
                        }
                    };

Minimum Path Sum

Algorithm

Recursive+memoisation Always take the min(dp[i-1][j],dp[i][j-1]) and it with the current element. 

                class Solution {
                    public:
                        int calc(vector<vector<int>> &grid,int m,int n,vector<vector<int>> &dp){
                            if(dp[m][n]!=-1)
                                return dp[m][n];
                            
                            if(m==0&&n==0)
                                return grid[m][n];
                            else if(m==0&&n!=0)
                            dp[m][n]= grid[m][n]+calc(grid,m,n-1,dp);
                             //return dp[m][n];}
                            else if(n==0&&m!=0){
                                dp[m][n]=grid[m][n]+calc(grid,m-1,n,dp);
                                //return dp[m][n];
                            }
                            else
                            {
                                dp[m][n]=grid[m][n]+min(calc(grid,m-1,n,dp),calc(grid,m,n-1,dp));
                            }
                            return dp[m][n];
                        }
                        int minPathSum(vector<vector<int>>& grid) {
                            int m=grid.size();
                            int n=grid[0].size();
                            vector<vector<int>> dp(m+1,vector<int> (n+1,-1));
                            return calc(grid,m-1,n-1,dp);
                        }
                    };

                class Solution {
                    public:
                        int minPathSum(vector<vector<int>>& grid) {
                            int n=grid.size();
                            int m=grid[0].size();
                            for(int i=1;i<n;i++){
                                grid[i][0]+=grid[i-1][0];
                            }
                            for(int j=1;j<m;j++){
                                grid[0][j]+=grid[0][j-1];
                            }
                            for(int i=1;i<n;i++){
                                for(int j=1;j<m;j++){
                                    grid[i][j]+=min(grid[i-1][j],grid[i][j-1]);
                                }
                            }
                            return grid[n-1][m-1];
                        }
                    };

Problem 25

Edit Distance

Algorithm

We have three option insertion,deletion and intertion.so while insertion it means that we have taken the particular index of the word2 string so we have to move to next index.while doing deletion it means we deleted the particular index of the string word1 and now we have to move further.while replacement we have to move further in both string. One important thing we must have to consider if the string word1 is empty what does it means it mean we have to insert the string word2 because we have nothing in string word1 so that we can replace it or we can delete it. it the word2 is empty it means that if we want to make particular string we have to delete the word1 because it is the only way so that our word1 string can match with word2. 

                class Solution {
                    public:
                        int minDistance(string word1, string word2) {
                            int n=word1.size();
                            int m=word2.size();
                            int dp[n+1][m+1];
                            for(int i=0;i<=n;i++){
                                for(int j=0;j<=m;j++){
                                    if(i==0&&j==0)dp[i][j]=0;
                                    else if(i==0)dp[i][j]=j;
                                    else if(j==0)dp[i][j]=i;
                                    else if(word1[i-1]==word2[j-1])dp[i][j]=dp[i-1][j-1];
                                    else dp[i][j]=min(dp[i-1][j],(min(dp[i-1][j-1],dp[i][j-1])))+1;
                                }
                            }
                            return dp[n][m];
                        }
                    };

Problem 26

Remove Nth Node From End of List

Algorithm

                class Solution {
                    public:
                        ListNode* removeNthFromEnd(ListNode* head, int n) {
                            ListNode *dummy=new ListNode(0);
                            dummy->next=head;
                            ListNode *slow=dummy,*fast=dummy;
                            int s=0,f=0;
                            while(fast!=NULL&&fast->next!=NULL){
                                slow=slow->next;
                                s++;
                                fast=fast->next->next;
                                f=f+2;
                            }
                            if(fast==NULL)f--;
                            int dis=f-n;
                            if(dis>s){
                            
                                int k=dis-s;
                                while(k--){
                                  slow=slow->next;  
                                }
                                slow->next=slow->next->next;
                            }
                            else {
                                
                                ListNode *temp=dummy;
                                while(dis--){
                                    temp=temp->next;
                                }
                                temp->next=temp->next->next;
                            }
                            
                               return dummy->next; 
                        }
                    };

Algorithm

Why we are taking dummy pointer? beacuse their are some case in which repiting number is started from the starting and available till end so in these case we have to remove the complete list. we use while loop because repiting number can be many number of times. 

                class Solution {
                    public:
                        ListNode* deleteDuplicates(ListNode* head) {
                            if(head==NULL)return head;
                            ListNode *dummy=new ListNode(-101);
                            dummy->next=head;
                            ListNode *prev=head,*temp=dummy;
                            head=head->next;
                            if(head==NULL)return dummy->next;
                            int flag=0;
                            while(head!=NULL){
                                while(head&&prev->val==head->val){
                                      head=head->next;
                                     flag=1;
                                }
                                    if(flag){
                                        temp->next=head;
                                        flag=0;}
                                    else
                                    temp=temp->next;
                                
                                    prev=head;
                                    if(head)head=head->next;
                            }
                            return dummy->next;
                        }
                    };

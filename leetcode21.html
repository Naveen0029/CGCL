Problem 90

Sort Colors

Algorithm

low point to zeros and high point to two. 

                class Solution {
                    public:
                        
                        void sortColors(vector<int>& nums) {
                            int low=0,mid=0,high=nums.size()-1;
                            while(mid<=high){
                                if(nums[mid]==0){
                                    swap(nums[mid],nums[low]);
                                    low++;
                                    mid++;
                                }
                                else if(nums[mid]==2){
                                    swap(nums[mid],nums[high]);
                                    high--;
                                }
                                else{
                                    mid++;
                                }
                            }
                        }
                    };

Problem 91

Partition Labels

Algorithm

we are starting from ending point.By using rfind STL we find the last repition of the index and we do this prcoes until we found our index then we come out from the loop and push the size in the vector. 

                class Solution {
                    public:
                        vector<int> partitionLabels(string S) {
                            vector<int> v;
                            int i=S.size()-1;
                            int siz=S.size()-1;
                            int max_index=i;
                            while(i>=0){
                                 while(i>=max_index){
                                     char str=S[i];
                                     int found=S.rfind(str,i-1);
                                     if(found!=string::npos){
                                         max_index=min(max_index,found);
                                         
                                     }
                                     i--;
                                 }
                                v.push_back((siz-max_index+1));
                                siz=max_index-1;
                                max_index=max_index-1;
                                i=max_index;
                            }
                            reverse(v.begin(),v.end());
                           return v;
                        }
                    };

Problem 92

Minimum Operations to Make the Array Increasing

Algorithm

if i+1th element is smaller than ith element then first set the size of i+1th element equal to ith element and then add 1.And also find the number of time we must have to do operation to make it strictly increasing. 

                class Solution {
                    public:
                        int minOperations(vector<int>& nums) {
                            int cnt=0;
                            for(int i=0;i<nums.size()-1;i++){
                                while(nums[i+1]<=nums[i]){
                                    
                                    cnt+=(nums[i]-nums[i+1])+1;
                                    nums[i+1]=nums[i]+1;
                                }
                            }
                            return cnt;
                        }
                    };

                class Solution {
                    public:
                        void dfs(vector<vector<char>>& board,int i,int j,int n,int m){
                            if(i<0||j<0||i>=n||j>=m||board[i][j]!='O')return;
                            
                            board[i][j]='G';
                            dfs(board,i+1,j,n,m);
                            dfs(board,i,j+1,n,m);
                            dfs(board,i-1,j,n,m);
                            dfs(board,i,j-1,n,m);
                            return;
                        }
                        void solve(vector<vector<char>>& board) {
                            int n=board.size();
                           
                            int m=0;
                            if(n!=0)m=board[0].size();
                            if(n==0&&m==0)return;
                            
                            
                            for(int i=0;i<n;i++){
                                if(board[i][0]=='O'){
                                    dfs(board,i,0,n,m);
                                }
                                if(board[i][m-1]=='O'){
                                     dfs(board,i,m-1,n,m);
                                }
                            }
                            for(int j=0;j<m;j++){
                                if(board[0][j]=='O'){
                                    dfs(board,0,j,n,m);
                                }
                                if(board[n-1][j]=='O'){
                                     dfs(board,n-1,j,n,m);
                                }
                            }
                            for(int i=0;i<n;i++){
                                for(int j=0;j<m;j++){
                                    if(board[i][j]=='G')board[i][j]='O';
                                    else board[i][j]='X';
                                }
                            }
                            return;
                        }
                    };

                class Solution {
                    public:
                        void dfs(int i,int j,vector<vector<char>>&board,vector<vector<char>>&mat,int n,int m){
                            if(i<0||i>=n||j<0||j>=m||mat[i][j]=='O'||board[i][j]=='X')return;
                            
                            mat[i][j]='O';
                            dfs(i+1,j,board,mat,n,m);
                            dfs(i,j+1,board,mat,n,m);
                            dfs(i-1,j,board,mat,n,m);
                            dfs(i,j-1,board,mat,n,m);
                        }
                        void solve(vector<vector<char>>& board) {
                            
                            int n=board.size();
                            int m=0;
                            if(n!=0)m=board[0].size();
                            if(n==0&&m==0)return;
                            vector<vector<char>> mat(n,vector<char>(m,'X'));
                            for(int i=0;i<n;i++){
                                if(board[i][0]=='O'&&mat[i][0]=='X'){
                                    dfs(i,0,board,mat,n,m);
                                }
                                if(board[i][m-1]=='O'&&mat[i][m-1]=='X'){
                                    dfs(i,m-1,board,mat,n,m);
                                }
                                
                            }
                            for(int j=0;j<m;j++){
                                if(board[0][j]=='O'&&mat[0][j]=='X'){
                                    dfs(0,j,board,mat,n,m);
                                }
                                if(board[n-1][j]=='O'&&mat[n-1][j]=='X'){
                                    dfs(n-1,j,board,mat,n,m);
                                }
                            }
                            for(int i=0;i<n;i++){
                                for(int j=0;j<m;j++){
                                    if(board[i][j]=='O'||board[i][j]=='X')
                                        board[i][j]=mat[i][j];
                                }
                            }
                        }
                    };

                class Solution {
                    public:
                        string decode(string &s,int &i){
                            string result="";
                            while(i<s.length()&&s[i]!=']'){
                                if(isdigit(s[i])){
                                    int k=0;
                                    while(i<s.length()&&isdigit(s[i])){
                                        k=k*10+s[i++]-'0';
                                    }
                                    i++;
                                    string r=decode(s,i);
                                    while(k--)result+=r;
                                    i++;
                                }else{
                                    result+=s[i++];
                                }
                            }
                            return result;
                        }
                        string decodeString(string s) {
                            int i=0;
                            return decode(s,i);
                        }
                    };

Domino and Tromino Tiling

Algorithm

So this is interesting question in which for n=3 the total number of ways are 5. How to think for the soln. first thing if there is 1*1 grid then how many ways to fill this ans the ans is 0. if their is 2*1 grid then number of ways is 1. if their is 2*2 grid then number of ways is 2.
0 1 2
1 2 5
so u are able to see the pattern. 

                class Solution {
                    public:
                        int mod=1000000007;
                        int numTilings(int N) {
                            if(N==1)return 1;
                            
                            int k=3;
                            vector<vector<unsigned int>>dp(k,vector<unsigned int>(N+1,0));
                            dp[1][1]=0;
                            dp[2][1]=1;
                            dp[1][2]=1;
                            
                                for(int j=2;j<=N;j++){
                                    for(int i=1;i<=2;i++){
                                        if(i==1&&dp[i][j]==0)dp[i][j]=((dp[i][j-1])%mod+(dp[i+1][j-2])%mod)%mod;
                                        else if(i!=1){
                                        dp[i][j]=((dp[i-1][j])%mod+(dp[i][j-1])%mod+(dp[i-1][j-1])%mod)%mod;
                                        
                                        }
                                    }
                                    
                                }
                            
                            return dp[2][N];
                        }
                    };

Problem 102

Target Sum

Algorithm

                class Solution {
                    public:
                        int Target(vector<int>& nums,int i,int cnt,int target,vector<vector<int>>&dp){
                            if(cnt==target&&(i==nums.size()))return 1;
                            else if(i>=nums.size())return 0;
                            
                            if(dp[i][cnt+1000]!=-1)return dp[i][cnt+1000];
                            
                            return dp[i][cnt+1000]=Target(nums,i+1,cnt+nums[i],target,dp)+Target(nums,i+1,cnt-nums[i],target,dp);
                        }
                        int findTargetSumWays(vector<int>& nums, int target) {
                            vector<vector<int>> dp(nums.size()+1,vector<int>(2001,-1));
                            return Target(nums,0,0,target,dp);
                        }
                    };

Algorithm

s1+s2=sum
s1-s2=target
2*s1=(target+sum)
so this is the variation of knapsack in which we have to find one subset only
[1,1,1,1,1] target=3
find all the subset whose total_sum is 4.
(5+3)/2=4. 

                class Solution {
                    public:
                        int findTargetSumWays(vector<int>& nums, int target) {
                            int sum=0;
                            for(int i=0;i<nums.size();i++){
                                sum+=nums[i];
                            }
                            if(sum<target||(sum+target)%2!=0)return 0;
                            int ft=(sum+target)/2;
                            int dp[nums.size()+1][ft+1];
                           
                            for(int i=0;i<=nums.size();i++){
                                for(int j=0;j<=ft;j++){
                                    if(j==0&&i==0)dp[i][j]=1;
                                    else if(i==0&&j>0)dp[i][j]=0;
                                    else if(nums[i-1]<=j){
                                        dp[i][j]=dp[i-1][j-nums[i-1]]+dp[i-1][j];
                                    }
                                    else dp[i][j]=dp[i-1][j];
                                }
                            }
                            return dp[nums.size()][ft];
                        }
                    };

Problem 103

Integer Break

Algorithm

we can say all the numbers can be maked from 2 to 9 so we make vector of number from 2 to 9 then apply knapsack.but it is unbounded knapsack becuase we can take the same number multiple times.
But their is an another approch in which we take all the numbers which is less than n. f(n)=k*(max(f(n-k),n-k)). 

  
                class Solution {
                    public:
                        void check(int sum,vector<int> &v,int res,int &f_r,int s){
                            if(sum==0){
                                f_r=max(f_r,res);
                                return;
                            }
                            if(sum<0||s<=0)return;
                            
                            if(v[s-1]<=sum){
                                check(sum-v[s-1],v,res*v[s-1],f_r,s);
                                check(sum,v,res,f_r,s-1);
                            }
                            else
                                check(sum,v,res,f_r,s-1);
                            
                            return;
                            
                        }
                        int integerBreak(int n) {
                            if(n==2)return 1;
                            if(n==3)return 2;
                            int f_r=INT_MIN;
                            int k=8;
                            vector<int> v(k);
                            int i=2;
                            int j=0;
                            while(i<10){
                                v[j++]=i;
                                i++;
                            }
                            
                            check(n,v,1,f_r,k);
                            return f_r;
                          
                        }
                    };

                class Solution {
                    public:
                        void Backtracking(vector<int>&nums,vector<vector<int>>&v,
                                          vector<int>&rv,int n,map<vector<int>,int>&mp,int cnt,vector<int> &temp){
                            if(cnt>=nums.size()){
                                if(mp.find(temp)==mp.end()){
                                    mp[temp]++;
                                    v.push_back(temp);
                                }
                                
                                
                                return;
                            }
                            for(int i=0;i<n;i++){
                                if(rv[i]==0){
                                    rv[i]=1;
                                    temp.push_back(nums[i]);
                                    Backtracking(nums,v,rv,n,mp,cnt+1,temp);
                                    temp.pop_back();
                                    rv[i]=0;
                                }
                            }
                            return ;
                            
                        }
                        vector<vector<int>> permuteUnique(vector<int>& nums) {
                            vector<int> rv(nums.size(),0);
                            vector<vector<int>> v;
                            map<vector<int>,int> mp;
                            vector<int> temp;
                            Backtracking(nums,v,rv,nums.size(),mp,0,temp);
                            return v;
                        }
                    };

                class Solution {
                    public:
                        void Backtracking(vector<int>&nums,vector<vector<int>>&v,
                                          vector<int>&rv,int n,map<vector<int>,int>&mp,int cnt,vector<int> &temp){
                            if(cnt>=nums.size()){
                                if(mp.find(temp)==mp.end()){
                                    mp[temp]++;
                                    v.push_back(temp);
                                }
                                return;
                            }
                            for(int i=0;i<n;i++){
                                if(rv[i]==0){
                                    if(i!=0&&!rv[i-1]&&nums[i]==nums[i-1])continue;
                                    
                                    rv[i]=1;
                                    temp.push_back(nums[i]);
                                    Backtracking(nums,v,rv,n,mp,cnt+1,temp);
                                    temp.pop_back();
                                    rv[i]=0;
                                }
                            }
                            return ;
                            
                        }
                        vector<vector<int>> permuteUnique(vector<int>& nums) {
                            vector<int> rv(nums.size(),0);
                            vector<vector<int>> v;
                            map<vector<int>,int> mp;
                            vector<int> temp;
                            sort(nums.begin(),nums.end());
                            Backtracking(nums,v,rv,nums.size(),mp,0,temp);
                            return v;
                        }
                    };

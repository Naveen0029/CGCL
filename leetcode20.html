Problem 86

Median of Two Sorted Arrays

Algorithm

length of array x must be eaual to length of array y. 

                class Solution {
                    public:
                       
                        double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
                            if(nums1.size()==0){
                                int i=(nums2.size()+1)/2;
                                if(nums2.size()%2==0){
                                     
                                    return double(nums2[i-1]+nums2[i])/2;
                                }
                                else return double(nums2[i-1]);
                            }
                              if(nums2.size()==0){
                                  int i=(nums1.size()+1)/2;
                                if(nums1.size()%2==0){
                                    
                                    return double(nums1[(i-1)]+nums1[i])/2;
                                }
                                else return double(nums1[i-1]);
                            }
                            if(nums1.size()>nums2.size())return findMedianSortedArrays(nums2,nums1);
                            
                            int low=0;
                            int high=nums1.size();
                            
                            while(low<=high){
                                int partitionX=(low+high)/2;
                                int partitionY=(nums1.size()+nums2.size()+1)/2-(partitionX);
                                
                                int max_leftX=partitionX>0?nums1[partitionX-1]:INT_MIN;
                                int min_rightX=partitionX<nums1.size()?nums1[partitionX]:INT_MAX;
                                
                                int max_leftY=partitionY>0?nums2[partitionY-1]:INT_MIN;
                                int min_rightY=partitionY<nums2.size()?nums2[partitionY]:INT_MAX;
                                
                                if((max_leftX<=min_rightY)&&(max_leftY<=min_rightX)){
                                    if((nums1.size()+nums2.size())%2==0)
                                        return double(max(max_leftX,max_leftY)+min(min_rightX,min_rightY))/2;
                                    else return double(max(max_leftX,max_leftY));
                                }
                                
                                else if(max_leftX>min_rightY){
                                    high=partitionX-1;
                                }
                                else if(max_leftY>min_rightX){
                                    low=partitionX+1;
                                }
                                
                            }
                           
                            return -1;
                        }
                    };

Problem 87

Find First and Last Position of Element in Sorted Array

Algorithm

when left is true and mid value equals to target it means that we have to search in the left subarray. 

                class Solution {
                    public:
                        int check(int low,int high,vector<int> &nums,int target,bool left){
                            
                            while(low<=high){
                                int mid=low+(high-low)/2;
                               
                               if(nums[mid]>target||(left&&nums[mid]==target))high=mid-1;
                               else low=mid+1;
                            }
                            return low;
                                
                        }
                        vector<int> searchRange(vector<int>& nums, int target) {
                            if(nums.size()==0)return{-1,-1};
                            
                            int n=2;
                            vector<int> v(n,-1);
                            int start=check(0,nums.size()-1,nums,target,true);
                            
                            if(start<nums.size()&&nums[start]==target)v[0]=start;
                            else return {-1,-1};
                            
                            int end=check(start+1,nums.size()-1,nums,target,false)-1;
                            v[1]=end;
                            
                            return v;
                            
                        }
                    };

1_0_0_0_1_0_0_0_0_0_0_0_0_0_0_0_0_0_0_1=ate,tea,eat.
 

                class Solution {
                    public:
                        void check(vector<string> &strs,map<string,vector<string>> &mp){
                            for(int i=0;i<strs.size();i++){
                                string temp=strs[i];
                                sort(temp.begin(),temp.end());
                                mp[temp].push_back(strs[i]);
                            }
                            return ;
                        }
                        vector<vector<string>> groupAnagrams(vector<string>& strs) {
                            if(strs.size()==0)return {{}};
                            map<string,vector<string>> mp;
                            check(strs,mp);
                            vector<vector<string>> v;
                            for(auto itr=mp.begin();itr!=mp.end();itr++){
                                v.push_back(itr->second);
                            }
                            return v;
                        }
                    };

Problem 89

Jump Game

Algorithm

This is good question. Brute Force approch- If you want to see the recursive solution then go to solution page of leetcode. In this approch we are setting true all the path that the person can visit and at last if we find the last index as visited(true) then return true. it can be further more optimize- whenever we find the last index visited our loop breaks. 

                class Solution {
                    public:
                        bool canJump(vector<int>& nums) {
                            if(nums[0]==0&&nums.size()>1)return false;
                            else if(nums[0]==0&&nums.size()==1) return true;
                            
                            vector<int> v(nums.size(),0);
                            v[0]=1;
                            for(int i=0;i<nums.size()-1;i++){
                               if(v[i]==1){
                                    int val=nums.size()<(nums[i]+i+1)?nums.size():nums[i]+i+1;
                                   for(int j=i+1;j<val;j++){
                                       v[j]=1;
                                   }
                                 }
                                else return false;
                         }
                            
                        if(v[nums.size()-1]==0)return false;
                        else return true;
                        }
                    };

Algorithm

we are making the last index visited because we are starting from the last index in this we changing our target beacause if from any index we can reach to the target then this index is our new target because any one coming to this index can go to the last index. so we are just checking if target<=i+nums[i].

                    public:
                    bool canJump(vector<int>& nums) {
                        if(nums[0]==0&&nums.size()>1)return false;
                        else if(nums[0]==0&&nums.size()==1) return true;
                        
                        vector<int> v(nums.size(),0);
                        v[nums.size()-1]=1;
                        int target=nums.size()-1;
                        for(int i=v.size()-1;i>=0;i--){
                            if(v[i]==1)continue;
                             else if(v[target]==1&&target<=i+nums[i])
                             {
                                 v[i]=1;
                                 target=i;
                             }
                            else if(v[i+nums[i]]==0) v[i]=0;
                        }
                        if(v[0]==1)return true;
                        else return false;
                    }
                };

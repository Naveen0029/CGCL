Problem 45

Populating Next Right Pointers in Each Node II

Algorithm

                
class Solution {
    public:
        Node* connect(Node* root) {
            if(root==NULL)
                return root;
            
            Node *root1=root->next;
            while(root1!=NULL&&root1->next!=NULL&&root1->left==NULL&&root1->right==NULL){
                root1=root1->next;
            }
            if(root->right!=NULL&&root->next!=NULL){
                if(root1->left!=NULL)
                    root->right->next=root1->left;
                else if(root1->right!=NULL)
                    root->right->next=root1->right;
                else
                    root->right->next=NULL;
            }
            else if(root->left!=NULL&&root->next!=NULL){
                if(root1->left!=NULL)
                    root->left->next=root1->left;
                else if(root1->right!=NULL)
                    root->left->next=root1->right;
                else
                    root->left->next=NULL;
            }
            else if(root->left==NULL&&root->right==NULL){
                return root;
            }
            
            
            if(root->left!=NULL&&root->right!=NULL)
                root->left->next=connect(root->right);
            else
                connect(root->right);
            
            connect(root->left);
            return root;
        }
    };

Problem 47

House Robber

Algorithm

                class Solution {
                    public:
                        int check(vector<int>& nums,int n,vector<int> &dp){
                            if(n<=0)return 0;
                           if(dp[n-1]!=-1)return dp[n-1];
                            
                            dp[n-1]=max(nums[n-1]+check(nums,n-2,dp),check(nums,n-1,dp));
                            return dp[n-1];
                        }
                        int rob(vector<int>& nums) {
                            vector<int> dp(nums.size()+1,-1);
                         return check(nums,nums.size(),dp);  
                        }
                    };

Reverse Linked List II

Algorithm

Reverse between the left and right. 

                class Solution {
                    public:
                        ListNode* reverseBetween(ListNode* head, int m, int n) {
                            ListNode *temp,*curr,*prev=NULL;
                            ListNode *start,*prev1=NULL;
                            ListNode *dummy=new ListNode(0);
                            dummy->next=head;
                            curr=dummy;
                            int k=m;
                            while(k--){
                                prev1=curr;
                                curr=curr->next;
                            }
                            start=curr;
                            int s=n-m+1;
                            while(s--&&curr!=NULL){
                                temp=curr->next;
                                curr->next=prev;
                                prev=curr;
                                curr=temp;
                            }
                            if(prev1!=NULL)
                            prev1->next=prev;
                            
                            start->next=curr;
                            
                            return dummy->next;
                        }
                    };

class Solution {
public:
    ListNode* rotateRight(ListNode* head, int k) {
        if(k==0||head==NULL)
            return head;
        
     ListNode* temp=head,*def=head;
        int count=1;
        while(temp->next!=NULL){
            temp=temp->next;
            count++;
        }
        if(k>=count)
            k=k%count;
        
        if(k==0)
            return head;
        
        int s=count-k-1;
        while(def!=NULL&&s--){
            def=def->next;
        }
        temp->next=head;
        head=def->next;
        def->next=NULL;
        
        return head;
    }
};

                            }

                        }

Problem 62

Letter Combinations of a Phone Number

Algorithm

Try to make tree like      
                                    -
                          a         b        c 
                    d   e   f     d  e  f   d  e  f 
                - - -   so on

class Solution {
                public:
                    
                    void Backtracking(string &digits,string s,vector<string> &mp,vector<string> &v,int index){
                        if(digits.size()<=index)
                        {
                            v.push_back(s);
                            return;
                        }
                           
                        int l=digits[index]-'0';
                           string t=mp[l]; 
                            for(int j=0;j<t.size();j++){
                                s=s+t[j];
                                Backtracking(digits,s,mp,v,index+1);
                               // s=s.substr(0,s.size()-1);
                                s.erase(s.size()-1);
                            }
                        
                        return ;
                    }
                    vector<string> letterCombinations(string digits) {
                       if(digits.size()==0)
                           return {};
                       vector<string>mp= {"", "", "abc", "def", "ghi", "jkl", "mno", "pqrs", "tuv", "wxyz"}; 
                        vector<string> v;
                        Backtracking(digits,"",mp,v,0);
                        return v;
                    }
                };

Problem 63

Number of Closed Islands

Algorithm

Make the dfs call to the all 0s which are connected to the border 0's. 

                class Solution {
                    public:
                        void dfs(int i,int j,vector<vector<int>> &grid){
                            if(i<0||j<0||i>=grid.size()||j>=grid[0].size()||grid[i][j]==1) return ;
                            
                            grid[i][j]=1;
                            dfs(i-1,j,grid);
                            dfs(i,j-1,grid);
                            dfs(i+1,j,grid);
                             dfs(i,j+1,grid);
                            
                            return ;
                        }
                        int closedIsland(vector<vector<int>>& grid) {
                            int n=grid.size();
                            int m=grid[0].size();
                            for(int i=0;i<m;i++){
                                if(grid[0][i]==0){
                                    dfs(0,i,grid);
                                }
                                if(grid[n-1][i]==0){
                                    dfs(n-1,i,grid);
                                }
                            }
                             for(int i=0;i<n;i++){
                                if(grid[i][0]==0){
                                    dfs(i,0,grid);
                                }
                                if(grid[i][m-1]==0){
                                    dfs(i,m-1,grid);
                                }
                            }
                            int count=0;
                            for(int i=0;i<n;i++){
                                for(int j=0;j<m;j++){
                                    if(grid[i][j]==0){
                                          dfs(i,j,grid);
                                          count++;
                                    }     
                                }
                            }
                            return count;
                        }
                    };

Problem 64

Largest Number

Algorithm

SORT and cancatenate the string for checking order. check a+b>b+a or not. 

                class Solution {
                    public:
                        string largestNumber(vector<int>& nums) {
                            int n=nums.size();
                            vector<string> v;
                            for(int i=0;i<n;i++){
                                ostringstream str1;
                                str1<<nums[i];
                                string c=str1.str();
                                v.push_back(c);
                            }
                            sort(v.begin(),v.end(),[](string &a,string &b){
                               return a+b>b+a;
                            });
                            if(v[0]=="0")
                            return "0";
                            
                            string f="";
                            
                            for(int i=0;i<v.size();i++){
                                f=f+v[i];
                            }
                            return f;
                        }
                    };

Problem 65

Combination Sum

Algorithm

Using Dynamic Programming 

class Solution {
                public:
                     void check(vector<int>& candidates,int target,vector<vector<int>>&v,vector<int>rv,map<vector<int>,int>
                                              &mp,int n){
                         if(target>0&&n<=0)return ;
                         if(target==0){
                             if(mp.find(rv)==mp.end()){
                                 v.push_back(rv);
                                 mp[rv]++;
                             }
                             
                             return ;
                         }
                         
                         if(candidates[n-1]<=target){
                             rv.push_back(candidates[n-1]);
                             check(candidates,target-candidates[n-1],v,rv,mp,n);
                             rv.pop_back();
                             check(candidates,target,v,rv,mp,n-1);
                         }
                         else check(candidates,target,v,rv,mp,n-1);
                         
                         return ;
                     }
                    vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
                        map<vector<int>,int> mp;
                        vector<vector<int>> v;
                        vector<int> rv;
                        check(candidates,target,v,rv,mp,candidates.size());
                        return v;
                    }
                };

                class Solution {
                    public:
                        void check(vector<int> &candidates,int target,int n,vector<int> rv,vector<vector<int>> &v,int index){
                            if(target==0)
                            {
                                v.push_back(rv);
                                return ;
                            }
                            if(target<0)
                                return;
                            for(int i=index;i<n;i++){
                                rv.push_back(candidates[i]);
                                check(candidates,target-candidates[i],n,rv,v,i);
                                rv.pop_back();
                            }
                            return ;
                            
                        }
                        vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
                              if(target==0)
                                return {};
                            int n=candidates.size();
                              vector<vector<int>> v;
                             vector<int> rv;
                            check(candidates,target,n,rv,v,0);
                            return v;
                        }
                    };

Problem 66

Clone Graph

Algorithm

Using DFS 

                class Solution {
                    public:
                        Node* dfs(Node *node,map<Node*,Node*>&mp){
                            if(mp.find(node)!=mp.end())
                                return mp[node];
                            
                            mp[node]=new Node(node->val,{});
                            for(Node *itr : node->neighbors){
                                mp[node]->neighbors.push_back(dfs(itr,mp));
                            }
                            return mp[node];
                        }
                        Node* cloneGraph(Node* node) {
                            if(node==NULL)return node;
                            map<Node*,Node*>mp;
                            return dfs(node,mp);
                        }
                    };

                class Solution {
                    public:
                        Node* cloneGraph(Node* node) {
                            if(node==NULL)
                                return node;
                            map<Node*,Node*> mp;
                            queue<Node* >q;
                            int vali=node->val;
                          
                            q.push(node);
                            Node *head=new Node(vali,{});
                            mp[node]=head;
                            while(!q.empty()){
                                Node *p=q.front();
                               
                                q.pop();
                                for(Node* itr : p->neighbors){
                                    if(mp.find(itr)==mp.end()){
                                        mp[itr]=new Node(itr->val,{});
                                        q.push(itr);
                                        //mp[p->val]->neighbors.push_back();
                                    }
                                    mp[p]->neighbors.push_back(mp[itr]);
                                }
                            }
                           return head;
                        }
                    };

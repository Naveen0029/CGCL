Problem 41

Search in Rotated Sorted Array

Algorithm

In this approch first we find the pivot point and after that we find in which strictly increasing number it lies. 

class Solution {
                public:
                    int pivot(vector<int> &nums){
                        
                        int start=0;
                        int end=nums.size()-1;
                       
                        if(nums[start]<nums[end])return -1;
                        while(start<=end){
                              int mid=(start+end)/2;
                            if(nums[mid]>nums[mid+1])
                                return mid;
                            if(nums[start]>nums[mid]){
                                end=mid-1;
                            }
                            else start=mid+1;
                           
                           
                        }
                        return -1;
                    }
                    int binarySearch(vector<int> &nums,int target,int start,int end){
                        while(start<=end){
                            int mid=(start+end)/2;
                            if(nums[mid]>target){
                                end=mid-1;
                            }
                            else if(nums[mid]<target) start=mid+1;
                            else return mid;
                        } 
                        return -1;
                    }
                    int search(vector<int>& nums, int target) {
                        if(nums.size()==1){
                            if(nums[0]==target)return 0;
                            else return -1;
                        }
                
                        
                        int piv=pivot(nums);
                        
                        if(piv==-1)return binarySearch(nums,target,0,nums.size()-1);
                        if(nums[piv]==target)return piv;
                      
                       if(nums[0]<target)return binarySearch(nums,target,0,piv-1);
                        else if(nums[0]>target)return binarySearch(nums,target,piv+1,nums.size()-1);
                        else return 0;
                           
                        return -1;
                    }
                };

Problem 42

Counting Bits

Algorithm

First set all the number which is power of two.After that all the number can be formed by using most siginficant bit + (numbr-msb); 

class Solution {
                public:
                    vector<int> countBits(int num) {
                        if(num==0)return {0};
                        int s=num;
                        vector<int> v(s+1,0);
                        v[1]=1;
                        for(int i=2;i<=s;i*=2){
                            v[i]=1;
                        }
                        int msb=0;
                        for(int i=0;i<=num;i++){
                            if(v[i]==1)msb=i;
                            else v[i]=v[msb]+v[i-msb];
                        }
                        return v;
                    }
                };

Problem 43

Next Permutation

Algorithm

We just have to find the next greater element of the element which is smaller then the i+1th element.After finding we have to replace the numbers. Now reverse all the number after the ith element. 

                 class Solution {
                public:
                    void swap(vector<int> &nums,int i,int j){
                         int val=nums[i];
                                nums[i]=nums[j];
                                nums[j]=val;
                               
                    }
                    void nextPermutation(vector<int>& nums) {
                        if(nums.size()==1)return;
                        int n=nums.size();
                       
                        int start=0;
                        for(int i=n-2;i>=0;i--){
                            if(nums[i]>=nums[i+1])
                                continue;
                            else {
                                swap(nums,i,n-1);
                                start=i+1;
                                break;
                            }
                        }
                       
                          
                         int end=n-1;
                        while(start<end){
                           
                            swap(nums,start,end);
                            start++;
                            end--;
                        } 
                        
                      
                        return ;
                    }
                };

Problem 44

Populating Next Right Pointers in Each Node

Algorithm

preorder traversal 

class Solution {
                public:
                    Node* connect(Node* root) {
                        if(root==NULL)return root;
                        if(root->left)root->left->next=root->right;
                        if(root->next&&root->next->left){
                            if(root->right)root->right->next=root->next->left;
                        }
                        connect(root->right);
                        connect(root->left);
                        return root;
                    }
                };

Algorithm

class Solution {
    public:
        Node* connect(Node* root) {
            if(root==NULL)
                return root;
            
            if(root->right!=NULL&&root->next!=NULL){
                root->right->next=root->next->left;
                
            }
            else if(root->left==NULL&&root->right==NULL){
                if(root->next!=NULL)
                    return root;
                else{
                    root->next=NULL;
                    return root;
                }
            }
            root->left->next=connect(root->right);
            connect(root->left);
            return root;
        }
    };

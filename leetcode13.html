Problem 55

Lowest Common Ancestor of a Binary Search Tree

Algorithm

This is BST and as we know that in bst the lesser element lies on the left side and the greater element lies on the right side. so if element is lies in this range then this lowestCommonAncestor. 

                class Solution {
                    public:
                        TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
                            if(root==NULL)
                                return root;
                            
                            if(root->val>=p->val&&root->val<=q->val||root->val<=p->val&&root->val>=q->val)
                                return root;
                            
                            if(root->val>p->val&&root->val>q->val)
                            return lowestCommonAncestor(root->left,p,q);
                            else
                           return lowestCommonAncestor(root->right,p,q);
                            
                            
                        }
                    };

Problem 56

Minimum Number of Arrows to Burst Balloons

Algorithm

Sort according to their starting point. 

                class Solution {
                    public:
                        
                        int findMinArrowShots(vector<vector<int>>& points) {
                            
                            int n=points.size();
                            if(n==0)
                                return 0;
                            else if(n==1)
                                return 1;
                            
                            sort(points.begin(),points.end(),[](const vector<int> &a,vector<int> &b){
                                return a[0]<b[0];
                            });
                            int a=points[0][0];
                            int b=points[0][1];
                            int count=1;
                            for(int i=1;i<n;i++){
                                if(b>=points[i][0]){
                                    a=points[i][0];
                                    b=min(points[i][1],b);
                                }
                                else
                                {
                                    a=points[i][0];
                                    b=points[i][1];
                                    count++;
                                   
                                }
                            }
                            return count;
                            
                        }
                    };

Algorithm

Sort according to their ending point. 

                    class Solution {
                        public:
                            int findMinArrowShots(vector<vector<int>>& points) {
                                if(points.size()==0)return 0;
                                sort(points.begin(),points.end(),[](const vector<int>&a,
                                                                   vector<int>&b){
                                    return a[1]<b[1];
                                });
                                int count=1;
                                int val=points[0][1];
                                for(int i=1;i<points.size();i++){
                                    if(points[i][0]<=val){
                                        points[i][1]=val;
                                    
                                    }
                                    else{
                                        count++;
                                        val=points[i][1];
                                    }
                                }
                                return count;
                                    
                            }
                        };

Problem 57

Two Sum II

Algorithm

If the value is less then we have to move i pointer else j pointer. 

                class Solution {
                    public:
                        vector<int> twoSum(vector<int>& numbers, int target) {
                            int n=numbers.size();
                            vector<int> v;
                            int i=0;
                            int j=n-1;
                          while(i<j){
                              if(numbers[i]+numbers[j]==target)
                                    {
                                        v.push_back(i+1);
                                        v.push_back(j+1);
                                         break;
                                    }
                              else if(numbers[i]+numbers[j]<target)
                                  i++;
                              else j--;
                          }
                                    
                            return v;
                        }
                    };

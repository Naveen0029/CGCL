Problem 18

Remove Linked List Elements

Algorithm

First Node can also be the node that we want to remove.so we are using dummy pointer.Node can also be in queue so we use while loop to remove them. 

                class Solution {
                    public:
                        ListNode* removeElements(ListNode* head, int val) {
                            ListNode *dummy=new ListNode(0);
                            dummy->next=head;
                            ListNode *temp=head,*def=dummy;
                            while(temp!=NULL){
                                while(temp&&temp->val==val){
                                    def->next=temp->next;
                                    temp=temp->next;
                                }
                                
                                
                                if(def->next)def=def->next;
                                if(temp)temp=temp->next;
                            }
                            return dummy->next;
                        }
                    };

Problem 19

Count Primes

Algorithm

class Solution {
                public:
                    int countPrimes(int n) {
                    
                        vector<int>  arr(n,1);
                        int count=0;
                        for(int i=2;i<n;i++){
                            if(arr[i]==1){
                                count++;
                                int j;
                                for(j=2;j*i<n;j++){
                                    arr[i*j]=0;
                                }
                            }
                        }
                        return count;
                    }
                };

Problem 20

Maximum Subarray

Algorithm

                class Solution {
                    public:
                        int maxSubArray(vector<int>& nums) {
                            int n=nums.size();
                            int sum=nums[0];
                            int max_val=nums[0];
                            for(int i=1;i<n;i++){
                                sum=max(nums[i],sum+nums[i]);
                                max_val=max(sum,max_val);
                                }
                            
                            
                            return max_val;
                        }
                    };

Problem 21

Climbing Stairs

Algorithm

                class Solution {
                    public:
                        int calc(int n,vector<int> &dp,int i){
                            
                            
                            if(i>n)
                                return 0;
                            
                            if(i==n)
                            {
                               return 1;
                            }
                            
                            if(dp[i]!=-1)
                                return dp[i];
                             
                            
                            dp[i]=calc(n,dp,i+1)+calc(n,dp,i+2);
                            return dp[i];
                            
                        }
                        int climbStairs(int n) {
                            vector<int> dp(n+1,-1);
                            return calc(n,dp,0);
                        }
                    };

Algorithm

We can only take two steps n-1 and n-2th.so we started the loop from the i=2.because if i=1 then it goes out of bound. 

                class Solution {
                    public:
                        int climbStairs(int n) {
                            int dp[n+1];
                            dp[0]=1;
                            dp[1]=1;
                            for(int i=2;i<=n;i++){
                                dp[i]=dp[i-1]+dp[i-2];
                            }
                            return dp[n];
                        }
                    };

Problem 22

Unique Paths

Algorithm

if i and j=0 means in the first row then their is only one path so for all i==0 and j==0 we set dp[i][j]=1. 

                class Solution {
                    public:
                        int uniquePaths(int m, int n) {
                            int dp[m][n];
                            for(int i=0;i<m;i++){
                                for(int j=0;j<n;j++){
                                    if(i==0||j==0)
                                        dp[i][j]=1;
                                    else
                                    dp[i][j]=dp[i-1][j]+dp[i][j-1];
                                }
                            }
                            return dp[m-1][n-1];
                        }
                    };

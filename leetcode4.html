Problem 14

Non-overlapping Intervals

Algorithm

First we have to sort it according to their second number and after that we maintain a variable last_index which contain the v[i][1].so we can check that it is overlapping or not. 

                class Solution {
                    public:
                        int eraseOverlapIntervals(vector<vector<int>>& intervals) {
                            if(intervals.size()==0)return 0;
                            
                            int n=intervals.size();
                            int m=intervals[0].size();
                            sort(intervals.begin(),intervals.end(),[](vector<int> &a,vector<int> &b){
                                if(a[1]<b[1])return true;
                                else if(a[1]==b[1])return a[0]<b[0];
                                else return false;
                            });
                            int last_index=INT_MIN;
                            int count=0;
                            for(int i=0;i<n;i++){
                                if(last_index>intervals[i][0])
                                {
                                    count++;
                                    
                                }
                                else
                                last_index=intervals[i][1];
                                
                            }
                            return count;
                                
                        }
                    };

Problem 15

Symmetric Tree

Algorithm

using recursion. 

                class Solution {
                    public:
                        bool symetric(TreeNode* root1,TreeNode* root2){
                            if(root1==NULL&&root2==NULL)
                                return true;
                            if(root1==NULL&&root2!=NULL||root2==NULL&&root1!=NULL)return false;
                            
                            if(root1->val==root2->val&&symetric(root1->right,root2->left)&&symetric(root1->left,root2->right))
                                return true;
                            else 
                                return false;
                        }
                        bool isSymmetric(TreeNode* root) {
                            if(root==NULL)
                                return true;
                            
                            return symetric(root->left,root->right);
                            
                        }
                    };

Problem 16

Binary Tree Zigzag Level Order Traversal

Algorithm

Using two stack. 

                class Solution {
                    public:
                        vector<vector<int>> zigzagLevelOrder(TreeNode* root) {
                            if(root==NULL)return {};
                            
                            vector<vector<int>> vec;
                            vector<int> v;
                            stack<TreeNode*> s;
                            stack<TreeNode*> s1;
                            s.push(root);
                            while(!s.empty()||!s1.empty()){
                                while(!s.empty()){
                                    TreeNode *n =s.top();
                                    s.pop();
                                    if(n->left)s1.push(n->left);
                                    if(n->right)s1.push(n->right);
                                    
                                    
                                    v.push_back(n->val);  
                                }
                                if(!v.empty())vec.push_back(v);
                                
                                v.clear();
                                while(!s1.empty()){
                                    TreeNode *p=s1.top();
                                    s1.pop();
                                   
                                    if(p->right)s.push(p->right);
                                    if(p->left)s.push(p->left);
                                    
                                    
                                    v.push_back(p->val);
                                }
                                 if(!v.empty())vec.push_back(v);
                                
                                 v.clear();
                            }
                            return vec;
                        }
                    };

Problem 17

Find the Duplicate Number

Algorithm

As the values of array is lie between 1 to n+1.so we can take the value of array as index of the array and making the value of the indexes means that we already traverse the index and we are traversing it again it means their is repitive value which lead us to the same index again. 

                class Solution {
                    public:
                        int findDuplicate(vector<int>& nums) {
                            int i;
                            int n=nums.size();
                            for(i=0;i<n;i++){
                                
                                  if(abs(nums[abs(nums[i])-1])==-nums[abs(nums[i])-1])
                                    break;
                                else
                                   nums[abs(nums[i])-1]=-nums[abs(nums[i])-1];
                                
                            }
                            return abs(nums[i]);
                        }
                    };

Problem 51

Coin Change

Algorithm

Recursion + memoization.This is the question of unbounded knapsack.we can take same number many times. Base condition if amount=0 then we do not take any coin. so if we have coins assume we have 5 coins of different number but the amount that we have to choose is equal to zero.so we do not take any coin.so we return 0. second condition:-when our n=0 and amount is greater then zero then it means we do not have sufficient coin to make the desire amount.so we return maximum ways of making amount using those coins. 

                class Solution {
                    public:
                        int find(vector <int > &coins,vector <vector <int > > &dp,int amount,int n){
                            if(amount==0)return 0;
                            else if((n <=0&&amount >0)||amount <0)return 100001;
                            
                            if(dp[n-1][amount]!=-1)return dp[n-1][amount];
                            
                            if(coins[n-1] <=amount){
                                dp[n-1][amount]=min(1+find(coins,dp,amount-coins[n-1],n),find(coins,dp,amount,n-1));
                            }
                            else 
                                dp[n-1][amount]=find(coins,dp,amount,n-1);
                            
                            return dp[n-1][amount];
                            
                        }
                        int coinChange(vector <int >& coins, int amount) {
                            int n=coins.size();
                            vector <vector <int > > dp(n+1,(vector <int >(amount+1,-1)));
                            int res=find(coins,dp,amount,n);
                            if(res >=10001)return -1;
                            else return res;
                        }
                    };

                                            dp[i][j]=10001;
                                        else if(j==0)
                                            dp[i][j]=0;
                                        else if(coins[i-1] <=j){
                                            dp[i][j]=min(1+dp[i][j-coins[i-1]],dp[i-1][j]);
                                        }
                                        else
                                            dp[i][j]=dp[i-1][j];
                                    }
                                   
                                }
                                if(dp[n][amount] >=10001)
                                    return -1;
                                else

                            }

Problem 52

Merge Two Binary Trees

Algorithm

                class Solution {
                    public:
                        TreeNode* mergeTrees(TreeNode* root1, TreeNode* root2) {
                            if(root1==NULL&&root2==NULL)return NULL;
                            
                            if(root1&&root2)root1- >val+=root2- >val;
                            else if(!root1&&root2)return root2;
                            else if(root1&&!root2)return root1;
                            
                            root1- >left=mergeTrees(root1- >left,root2- >left);
                            root1- >right=mergeTrees(root1- >right,root2- >right);
                            return root1;
                        }
                    };

                class Solution {
                    public:
                        int change(int amount, vector <int >& coins) {
                             int n=coins.size();
                               int dp[n+1][amount+1];
                            for(int i=0;i <=n;i++){
                                for(int j=0;j <=amount;j++){
                                    if(i==0&&j >0)
                                        dp[i][j]=0;
                                    else if(j==0)
                                        dp[i][j]=1;
                                    else if(coins[i-1] <=j){
                                        dp[i][j]=dp[i-1][j]+
                                                    dp[i][j-coins[i-1]];
                                    }
                                    else
                                        dp[i][j]=dp[i-1][j];
                                }
                               
                            }
                         
                             return dp[n][amount];
                        }
                        
                    };

Problem 54

Subtree of Another Tree

Algorithm

Treat every node as root and check differently. 

                class Solution {
                    public:
                        bool check(TreeNode *s,TreeNode *t){
                            if(s==NULL&&t==NULL)return true;
                            else if(s==NULL||t==NULL)return false;
                            
                            if(s- >val==t- >val&&check(s- >left,t- >left)&&check(s- >right,t- >right))return true;
                            else return false;
                        }
                        bool isSubtree(TreeNode* s, TreeNode* t) {
                            if(s==NULL)return false;
                            
                            if(check(s,t)||isSubtree(s- >left,t)||isSubtree(s- >right,t))
                                return true;
                            else return false;
                        }
                    };

Problem 115

Find Median from Data Stream

Algorithm

Using Heaps.
Ref 1
 

                class MedianFinder {
                    public:
                        /** initialize your data structure here. */
                     
                         //use max_heap and min_heap
                        priority_queue<int> max_heap;
                        priority_queue<int,vector<int>,greater<int>> min_heap;
                      
                        MedianFinder() {
                        
                        }
                        
                        void addNum(int num) {
                             if(max_heap.empty()||max_heap.top()>num){
                                 max_heap.push(num);
                             }
                            else{
                                min_heap.push(num);
                            }
                            
                            
                            if(max_heap.size()>min_heap.size()+1){
                                min_heap.push(max_heap.top());
                                max_heap.pop();
                            }
                            else if(min_heap.size()>max_heap.size()+1){
                                max_heap.push(min_heap.top());
                                min_heap.pop();
                            }
                        
                        }
                        
                        double findMedian() {
                           if(max_heap.size()==min_heap.size())
                           return (double)((double)max_heap.top()+(double)min_heap.top())/2.0;
                           
                            if(max_heap.size()>min_heap.size())return (double)(max_heap.top());
                            else return (double)(min_heap.top());
                            
                        }
                    };

Problem 112

Maximum Product of Word Lengths

Algorithm

First create a vector whose every index contain bit value of the string.
Ref 1
 

                class Solution {
                    public:
                        int maxProduct(vector<string>& words) {
                            int ans=0;
                            vector<int> v;
                            for(int i=0;i<words.size();i++){
                                int val=0;
                                string s=words[i];
                                for(int j=0;j<s.size();j++){
                                    int bit=s[j]-'a';
                                    if(!((val>>bit)&1)){
                                        val+=1<<bit;
                                    }
                                }
                                v.push_back(val);
                            }
                            for(int i=0;i<v.size()-1;i++){
                                for(int j=i+1;j<v.size();j++){
                                    if((v[i]&v[j])==0){
                                        ans=max(ans,(int)(words[i].size()*words[j].size()));
                                    }
                                }
                            }
                            return ans;
                        }
                    };

Reorganize String

typedef pair<int,char> p;
class Solution {
public:
    string reorganizeString(string S) {
        string st="";
        map<char,int> mp;
        priority_queue<p> pq;
        for(auto str:S){
            mp[str]++;
        }
        for(auto itr=mp.begin();itr!=mp.end();itr++){
            pq.push(make_pair(itr->second,itr->first));
        }
        while(pq.size()>1){
            auto p1=pq.top();
            pq.pop();
            auto p2=pq.top();
            pq.pop();
            st+=p1.second;
            st+=p2.second;
            mp[p1.second]--;
            mp[p2.second]--;
            if(mp[p1.second]>0){
                pq.push(make_pair(mp[p1.second],p1.second));
            }
            if(mp[p2.second]>0){
                pq.push(make_pair(mp[p2.second],p2.second));
                
            }
        }
        if(pq.size()>0){
            if(pq.top().first>1)return "";
            else st+=pq.top().second;
        }
        
        
        return st;
    }
};

Problem 114

Smallest Range Covering Elements from K Lists

Algorithm

Using min heap
Ref 1
 

typedef pair<int,pair<int,int>> p;

class Solution {
public:
    vector<int> smallestRange(vector<vector<int>>& nums) {
        //maintain two heaps one is max and another is min
        
        vector<int> v;
        int res=INT_MAX;//for handling diff of range
        priority_queue<p,vector<p>,greater<p>> min_heap;
        int a=0,b=INT_MAX;//b is max_val because first we want it show max range 
        int max_val=INT_MIN;//it contain maximum value so that their is no need to use max_heap
        for(int i=0;i<nums.size();i++){
            
            min_heap.push(make_pair(nums[i][0],make_pair(i,0)));
            if(max_val<nums[i][0])max_val=nums[i][0];
        }
        while(min_heap.size()>=nums.size()){
            auto pt=min_heap.top();
            min_heap.pop();
             
            
            res=max_val-pt.first;
            if((max_val!=INT_MIN)&&(b-a)>(res)){
                a=pt.first;
                b=max_val;
            }
            else if(((b-a)==res)&&(a>(pt.first))){//work when range is equal
                a=pt.first;
                b=max_val;
            }
            int i=pt.second.first;
            int j=pt.second.second;
            if(j+1<nums[i].size()){//next element of the array whose minimum is taken
                max_val=max(max_val,nums[i][j+1]);
                min_heap.push(make_pair(nums[i][j+1],make_pair(i,j+1)));
            }
            
            
        }
        v.push_back(a);
        v.push_back(b);
        return v;
    }
};

Problem 80

The K Weakest Rows in a Matrix

Algorithm

count the number of soldiers and sort according to soldiers. 

                bool comp(pair<int,int>&a,pair<int,int>&b){
                    return a.second==b.second?a.first<b.first:a.second<b.second;
                }
                class Solution {
                public:
                    vector<int> kWeakestRows(vector<vector<int>>& mat, int k) {
                        int n=mat.size();
                        int m=mat[0].size();
                        int count;
                        map<int,int> mp;
                        for(int i=0;i<n;i++){
                            count=0;
                            for(int j=0;j<m;j++){
                                if(mat[i][j]==1)count++;
                            }
                            mp[i]=count;
                        }
                        vector<pair<int,int>> v;
                        for(auto &it:mp){
                            v.push_back(it);
                        }
                        sort(v.begin(),v.end(),comp);
                        vector<int> vec;
                        for(int i=0;i<k;i++){
                            vec.push_back(v[i].first);
                        }
                        return vec;
                    }
                };

Problem 81

Generate Parentheses

Algorithm

if closed Parentheses is greater then return. 

                class Solution {
                    public:
                        void Backtrack(int ob,int cb,string str,vector<string> &v){
                            if(ob>cb)return;
                            if(ob==0){
                                while(cb--)str+=")";
                                
                                v.push_back(str);
                                return;
                            }
                            Backtrack(ob-1,cb,str+"(",v);
                            Backtrack(ob,cb-1,str+")",v);
                            return ;
                        }
                        vector<string> generateParenthesis(int n) {
                            vector<string> v;
                            Backtrack(n,n,"",v);
                            return v;
                        }
                    };

Algorithm

                class Solution {
                    public:
                        void Backtrack(int n,int left,int right,vector<string> &v,string str){
                            if(left==right&&right==n){
                                v.push_back(str);
                                return ;
                            }
                            if(left<right)return;
                            if(left>n||right>n)return;
                            
                            Backtrack(n,left+1,right,v,str+'(');
                            Backtrack(n,left,right+1,v,str+')');
                            return ;
                            
                            
                        }
                        vector<string> generateParenthesis(int n) {
                            vector<string> v;
                            Backtrack(n,0,0,v,"");
                            return v;
                        }
                    };

Problem 82

Permutations

Algorithm

Take a visited array so that when the loop is running from index 0 and if the particluar index is 1 then it is taken we just have to ignore it and move further. For the base we can take the s and we can also say that if the size of the rv vector is equal to the size of nums array then this is the condition of termination. 

                class Solution {
                    public:
                        void Backtrack(vector<int> &nums,vector<vector<int>> &v,vector<int> rv,vector<bool> &visited,int s){
                            if(s>=nums.size()){
                                v.push_back(rv);
                                return;
                            }
                            for(int i=0;i<nums.size();i++){
                                if(!visited[i]){
                                rv.push_back(nums[i]);
                                visited[i]=true;
                                Backtrack(nums,v,rv,visited,s+1);
                                visited[i]=false;
                                rv.pop_back();
                                }
                               
                            }
                            return ;
                        }
                        vector<vector<int>> permute(vector<int>& nums) {
                            vector<int> rv;
                            vector<vector<int>> v;
                            vector<bool> visited(nums.size(),false);
                            Backtrack(nums,v,rv,visited,0);
                            return v;
                        }
                    };

Problem 83

Perfect Squares

Algorithm

First make the vector of the perfect square which is less than given number. After that it is just simple knapsack. 

                class Solution {
                    public:
                        int solve(vector<int> &v,int n,int target,vector<vector<int>> &dp){
                            if(target==0)
                            {
                             return 0;
                            }
                            else if(n==0&&target>0){
                                
                                return 10000 ;
                            }
                            if(dp[n-1][target]!=-1)return dp[n-1][target];
                            
                            if(v[n-1]<=target&&v[n-1]!=0)
                                 dp[n-1][target]=min(1+solve(v,n,target-v[n-1],dp),solve(v,n-1,target,dp));
                            
                            else
                                dp[n-1][target]=solve(v,n-1,target,dp);
                                
                            return dp[n-1][target];
                                
                        }
                        int numSquares(int n) {
                            vector<int> v;
                            for(int i=1;i*i<=n;i++){
                                v.push_back(i*i);
                            }
                         
                            vector<vector<int>>dp(v.size()+1,vector<int>(n+1,-1));
                            return solve(v,v.size(),n,dp);
                            
                        }
                    };

Algorithm

so at the place of 1+table[j-v[i]] we have another choice to use table[v[i]]+table[j-v[i]] but for this first we have to mark the index(perfect square) with 1 and this can be done while we are making vector of perfect square .we just say that table[i*i]=1. it give more feal then 1+table[j-v[i]] becaue for making 8. 8=4+(8-4); 

                    class Solution {
                        public:
                            
                            int numSquares(int n) {
                                vector<int> v;
                                for(int i=1;i*i<=n;i++){
                                    v.push_back(i*i);
                                }
                                vector<int> table(n+1,INT_MAX);
                                table[0]=0;
                                for(int i=0;i<v.size();i++){
                                    for(int j=v[i];j<=n;j++){
                                        table[j]=min(table[j],1+table[j-v[i]]);
                                    }
                                }
                                return table[n];
                            }
                        };

Problem 84

Word Break

Algorithm

Recursion+memo. 

                class Solution {
                    public:
                        map<string,bool> mp;
                        bool finde(string s,vector<string> &wordDict){
                            for(int i=0;i<wordDict.size();i++){
                                if(s==wordDict[i])
                                    return true;
                            }
                            return false;
                        }
                        bool wordBreak(string s, vector<string>& wordDict) {
                            if(s.size()==0)return true;
                            
                            if(mp.find(s)!=mp.end())
                                return mp[s];
                            
                            for(int i=s.size();i>=1;i--){
                                if(finde(s.substr(0,i),wordDict)&&wordBreak(s.substr(i,s.size()-i),wordDict)){
                                    mp[s.substr(0,i)]=true;
                                    return true;
                                }
                            }
                            mp[s]=false;
                            return false;
                        }
                    };

                            vector<bool> dp(s.size()+1,false);
                            dp[0]=true;
                            for(int i=0;i<=s.size();i++){
                                for(int j=0;j<i;j++){
                                    if(finde(s.substr(j,i-j),wordDict)&&dp[j]){
                                    dp[i]=true;
                                }
                                }
                            }
                             return dp[s.size()];   
                        }

Problem 85

Add Two Numbers

Algorithm

Recursion 

                class Solution {
                    public:
                        void makeList(ListNode* l1,ListNode *l2,ListNode *head,int &carry){
                            if(!l1&&!l2)return ;
                            
                            if(l1&&l2){
                               
                                int sum=l1->val+l2->val+carry;
                                if(sum>=10){
                                    sum=sum%10;
                                    carry=1;
                                }
                                else carry=0;

                                head->next=new ListNode(sum);
                                makeList(l1->next,l2->next,head->next,carry);
                            }
                            else if(l1&&!l2){
                                int sum=l1->val+carry;
                                if(sum>=10){
                                    sum=sum%10;
                                    carry=1;
                                }
                                else carry=0;
                        
                                head->next=new ListNode(sum);
                               makeList(l1->next,l2,head->next,carry); 
                            }
                            else{
                                int sum=l2->val+carry;
                                if(sum>=10){
                                    sum=sum%10;
                                    carry=1;
                                }
                                else carry=0;
            
                                head->next=new ListNode(sum);
                               makeList(l1,l2->next,head->next,carry); 
                            } 
                            
                            return ;
                                
                        }
                        ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
                            ListNode *dummy=new ListNode(0);
                            
                            ListNode *head=dummy;
                            int carry=0;
                            makeList(l1,l2,dummy,carry);
                            if(carry){
                                while(head->next)head=head->next;
                                
                                head->next=new ListNode(carry);
                            }
                            return dummy->next;
                        }
                    };

                class Solution {
                    public:
                        ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
                            ListNode *dummy=new ListNode(0);
                            ListNode *head=dummy;
                            int carry=0;
                            int data;
                            while(l1||l2){
                                if(l1&&l2){
                                    data=l1->val+l2->val+carry;
                                    head->next=new ListNode(data%10);
                                    head=head->next;
                                    carry=(data>=10?1:0);
                                    
                                }
                                else if(l1&&!l2){
                                     data=l1->val+carry;
                                    head->next=new ListNode(data%10);
                                    head=head->next;
                                    carry=(data>=10?1:0);
                                }
                                else{
                                    data=l2->val+carry;
                                    head->next=new ListNode(data%10);
                                    head=head->next;
                                    carry=(data>=10?1:0);
                                }
                                if(l1)l1=l1->next;
                                if(l2)l2=l2->next;
                            }
                            if(carry>0)head->next=new ListNode(carry);

Problem 71

Find the Town Judge

Algorithm

calculate the in and out degree.if in degree==n-1 and outdegree is equal to zero return that index. 

                class Solution {
                    public:
                        int findJudge(int N, vector<vector<int>>& trust) {
                            if(trust.size()==0&&N==1)
                                return 1;
                            
                            vector<vector<int>> v(N+1);
                            for(int i=0;i<trust.size();i++){
                                v[trust[i][0]].push_back(trust[i][1]);
                            }
                            map<int,int> mp;
                            for(int i=0;i<trust.size();i++){
                                mp[trust[i][1]]++;
                            }
                            int min_value=INT_MIN;
                            int index=0;
                            for(auto itr=mp.begin();itr!=mp.end();itr++){
                                if(itr->second>min_value)min_value=itr->second,index=itr->first;
                                
                            }
                            if(v[index].size()!=0||min_value!=v.size()-2||trust.size()==0)
                                return -1;
                            else return index;
                        }
                    };

Problem 72

All Paths From Source to Target

Algorithm

                class Solution {
                    public:
                        
                        void dfs(int i,vector<vector<int>> &graph,vector<int>rv,vector<vector<int>> &v){
                            rv.push_back(i);
                            for(auto itr:graph[i]){
                                dfs(itr,graph,rv,v);
                            }
                            if(i==graph.size()-1)
                                v.push_back(rv);
                            
                            return ;
                        }
                        vector<vector<int>> allPathsSourceTarget(vector<vector<int>>& graph) {
                            vector<int> rv;
                            vector<vector<int>> v;
                            dfs(0,graph,rv,v);
                            return v;
                        }
                    };

Problem 73

Count Negative Numbers in a Sorted Matrix

Algorithm

                class Solution {
                    public:
                        int isSafe(int i,int j,int m,int n){
                            if(i<0||i>=m||j<0||j>=n)return 0;
                            
                            return 1;
                        }
                        int countNegatives(vector<vector<int>>& grid) {
                            int m=grid.size();
                            int n=grid[0].size();
                            if(grid[m-1][n-1]>=0)
                                return 0;
                            
                            int count=0;
                            queue<pair<int,int>> q;
                            q.push(make_pair(m-1,n-1));
                            while(!q.empty()){
                                pair<int,int> p=q.front();
                                q.pop();
                                if(isSafe(p.first-1,p.second,m,n)&&grid[p.first-1][p.second]<0){
                                    q.push(make_pair(p.first-1,p.second));
                                    grid[p.first-1][p.second]=0;
                                }
                                if(isSafe(p.first,p.second-1,m,n)&&grid[p.first][p.second-1]<0){
                                    q.push(make_pair(p.first,p.second-1));
                                    grid[p.first][p.second-1]=0;
                                }
                                /*if(isSafe(p.first+1,p.second,m,n)&&grid[p.first+1][p.second]<0){
                                    q.push(make_pair(p.first+1,p.second));
                                    grid[p.first+1][p.second]=0;
                                }
                                 if(isSafe(p.first,p.second+1,m,n)&&grid[p.first][p.second+1]<0){
                                    q.push(make_pair(p.first,p.second+1));
                                    grid[p.first][p.second+1]=0;
                                }*/
                                count++;    
                            }
                            return count;
                        }
                    };

Algorithm

                class Solution {
                    public:
                        int countNegatives(vector<vector<int>>& grid) {
                            int m=grid.size()-1;
                            int n=grid[0].size()-1;
                            if(grid[m][n]>0)
                                return 0;
                            
                            int count=0;
                            while(m>=0&&grid[m][n]<0){
                                for(int j=n-1;j>=0;j--){
                                    if(grid[m][j]<0)count++;
                                    else break;
                                }
                                count++;
                                m--;
                            }
                            return count;
                        }
                    };

Problem 74

Merge Two Sorted Lists

Algorithm

                class Solution {
                    public:
                        ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
                            ListNode *dummy=new ListNode(0);
                            ListNode *head=dummy;
                            while(l1!=NULL&&l2!=NULL){
                                if(l1->val<=l2->val){
                                    head->next=l1;
                                    head=head->next;
                                    l1=l1->next;
                                }
                                else{
                                    head->next=l2;
                                    head=head->next;
                                    l2=l2->next;
                                }
                            }
                            while(l1!=NULL){
                                head->next=l1;
                                head=head->next;
                                l1=l1->next;
                            }
                            while(l2!=NULL){
                                head->next=l2;
                                head=head->next;
                               l2=l2->next;
                            }
                            return dummy->next;
                        }
                    };

Problem 58

Longest Substring with At Least K Repeating Characters

Algorithm

                class Solution {
                    public:
                        int check(string s,int start,int end,char c){
                            for(int i=start;i<=end;i++)
                                if(s[i]==c)
                                    return i;
                               
                            return 0;
                        }
                        int longest(string s,int start,int end,int k){
                            if(start>=end)
                                return INT_MIN;
                            
                            map<char,int> mp;
                            for(int i=start;i<=end;i++){
                                mp[s[i]]++;
                            }
                            int flag=0;
                            int index;
                            for(auto itr=mp.begin();itr!=mp.end();itr++){
                                if(itr->second<k)
                                {
                                    flag=1;
                                    char c=itr->first;
                                    index=check(s,start,end,c);
                                }
                            }
                            if(flag==0)
                                return end-start+1;
                            
                          
                            return max(longest(s,start,index-1,k),longest(s,index+1,end,k));
                        }
                        int longestSubstring(string s, int k) {
                            if(k==1)
                                return s.size();
                            
                            int d=longest(s,0,s.size()-1,k);
                            if(d==INT_MIN)
                            return 0;
                            else return d;
                        }
                    };

Problem 59

Number of Islands

Algorithm

DFS 

                class Solution {
                    public:
                        void dfs(int i,int j,int n,int m,vector<vector<char>> &grid){
                            if(i<0||j<0||i>=n||j>=m)
                                return;
                            
                            if(grid[i][j]=='0')
                                return;
                            
                            grid[i][j]='0';
                            dfs(i+1,j,n,m,grid);
                            dfs(i-1,j,n,m,grid);
                            dfs(i,j+1,n,m,grid);
                            dfs(i,j-1,n,m,grid);
                            
                            return;
                        }
                        int numIslands(vector<vector<char>>& grid) {
                            int n=grid.size();
                            int m=grid[0].size();
                            int count=0;
                            for(int i=0;i<n;i++){
                                for(int j=0;j<m;j++){
                                    if(grid[i][j]=='1')
                                    {
                                        dfs(i,j,n,m,grid);
                                        count++;
                                    }
                                }
                            }
                            return count;
                        }
                    };

                class Solution {

                            if(index>=s.size())
                                return ;

                                int ascii=s[index];

                                 Backtracking(s,index+1,v);

                class Solution {
                    public:
                        void Backtrack(string s,string str,vector<string>&v){
                            if(s.size()==0){
                                v.push_back(str);
                                return ;
                            }
                            string k=s.substr(1);
                            if(s[0]>='a'&&s[0]<='z'){
                                Backtrack(k,str+s[0],v);
                                Backtrack(k,str+char(s[0]-32),v);
                            }
                            else if(s[0]>='A'&&s[0]<='Z'){
                                Backtrack(k,str+s[0],v);
                                Backtrack(k,str+char(s[0]+32),v);
                            }
                            else{
                                Backtrack(k,str+s[0],v);
                            }
                        }
                        vector<string> letterCasePermutation(string S) {
                            vector<string> v;
                            Backtrack(S,"",v);
                            return v;
                         }
                    };

class Solution {
                public:
                    int leastInterval(vector<char>& tasks, int n) {
                        int l=tasks.size();
                        if(n==0||l==1)
                            return l;
                    
                        map<char,int> mp;
                        for(int i=0;i<l;i++){
                            mp[tasks[i]]++;
                        }
                        
                        int max_freq=INT_MIN;
                        for(auto itr=mp.begin();itr!=mp.end();itr++){
                          
                            max_freq=max(max_freq,itr->second);
                        }
                        int count=(max_freq-1)*(n+1);
                       
                       for(auto it=mp.begin();it!=mp.end();it++){
                           if(max_freq==it->second)
                               count++;
                       }
                        return max(l,count);
                    }
                };

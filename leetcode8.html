Problem 32

Minimum Operations to Make Array Equal

Algorithm

Their is simple logic if one value is increase then another one is decreased. First we have to calculate average. Here average of these number is showing that all the number become equal to their average. 

                class Solution {
                    public:
                        int minOperations(int n) {
                            int x=0;
                            for(int i=0;i<n;i++)x+=(2*i+1);
                            x/=n;
                            int ans=0;
                            for(int i=0;i<n/2;i++){
                                ans=ans+(x-(2*i+1));
                            }
                            return ans;
                        }
                    };

Problem 33

Matrix Block Sum

Algorithm

Brute Force approch 

                class Solution {
                    public:
                        int setMatrix(int i,int j,int n,int m,int k,vector<vector<int>>& mat){
                            int sum=0;
                            for(int s=max(0,i-k);s<=min(n,i+k);s++){
                                for(int r=max(0,j-k);r<=min(m,j+k);r++){
                                    sum=sum+mat[s][r];
                                }
                            }
                            return sum;
                        }
                        vector<vector<int>> matrixBlockSum(vector<vector<int>>& mat, int K) {
                            int n=mat.size();
                            int m=mat[0].size();
                            
                            vector<vector<int>> answer(n,(vector<int> (m,0)));
                            for(int i=0;i<n;i++){
                                for(int j=0;j<m;j++){
                                    answer[i][j]=setMatrix(i,j,n-1,m-1,K,mat);
                                }
                            }
                            return answer;
                        }
                    };

Problem 34

Single Number

Algorithm

a^a=0 and 0^a=a 

class Solution {
                public:
                    int singleNumber(vector<int>& nums) {
                        int ans=nums[0];
                        for(int i=1;i<nums.size();i++){
                            ans=ans^nums[i];
                        }
                        return ans;
                    }
                };

Algorithm

(a+b+c)-(a+b) 

                class Solution {
                    public:
                        int singleNumber(vector<int>& nums) {
                            map<int,int> mp;
                            int a=0;
                            int b=0;
                            
                            for(int i=0;i<nums.size();i++){
                                if(mp.find(nums[i])==mp.end()){
                                    a+=nums[i];
                                    mp[nums[i]]++;
                                }
                                else b+=nums[i];
                            }
                            return a-b;
                        }
                    };

Problem 35

Container With Most Water

Algorithm

We take two pointers one on the starting position and the other one on the ending positiion. The logic behind the taking of two pointer is that the height which is smaller always matters because if one side height is bigger and other side height is smaller then the maximum water store depends on the minimum height. Now how two pointers work.when we get the smaller height on one side then we have to move the pointer to the next position (end-1)or(start+1).because we always take one side maximum so we must have to move the sorter side. 

                class Solution {
                    public:
                        int maxArea(vector<int>& height) {
                            int start=0;
                            int end=height.size()-1;
                            int area=0;
                            while(start<end){
                                int value_min=min(height[start],height[end]);
                                area=max(area,value_min*(end-start));
                                if(height[start]<height[end])
                                    start++;
                                else end--;
                                
                                if(start>=end)
                                    break;
                                
                            }
                            return area;
                        }
                    };

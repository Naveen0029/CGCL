Problem 28

Linked List Cycle II

Algorithm

This code successfully run if u set tortoise and hare equal to head.Their is no need to take dummy pointer. 

                class Solution {
                    public:
                        ListNode *detectCycle(ListNode *head) {
                            if(head==NULL)return {};
                        
                             ListNode *dummy=new ListNode(-1);
                             dummy->next=head;
                             ListNode *tortoise=dummy,*hare=dummy;
                             do{
                                tortoise=tortoise->next;
                                if(hare&&hare->next)hare=hare->next->next;
                                else return NULL;
                                
                             }while(hare&&tortoise!=hare);
                            
                             if(hare==NULL)return NULL;
                             tortoise=dummy;
                             while(tortoise!=hare){
                                tortoise=tortoise->next;
                                 hare=hare->next;    
                            }
                            return tortoise;
                        }
                    };

Problem 29

Copy List with Random Pointer

Algorithm

Recursive 

                unordered_map<Node*,Node*> mp;
                class Solution {
                public:
                    Node* copyRandomList(Node* head) {
                        if(head==NULL)
                            return {};
                        
                        Node *root=new Node(head->val);
                        mp[head]=root;
                        
                        root->next=copyRandomList(head->next);
                        
                        if(head->random!=NULL)
                        root->random=mp[head->random];
                        else
                            root->random=NULL;
                        
                        return root;
                    }
                };

Algorithm

Iterative 

                class Solution {
                    public:
                        Node* copyRandomList(Node* head) {
                            if(head==NULL)return head;
                            
                            unordered_map<Node*,Node*> mp;
                            Node *temp=head;
                            while(temp!=NULL){
                                mp[temp]=new Node(temp->val);
                                temp=temp->next;
                            }
                            temp=head;
                            while(temp!=NULL){
                                mp[temp]->next=mp[temp->next];
                                mp[temp]->random=mp[temp->random];
                                temp=temp->next;
                            }
                            return mp[head];
                        }
                    };

Problem 30

Longest Substring Without Repeating Characters

Algorithm

we use the sliding approch in which when we find the same element in newly formed string which lies between the 0 to index-1,so we remove the string upto that particular index which lies between 0 to index-1.
Using inbuilt function for finding string. 

                class Solution {
                    public:
                        int lengthOfLongestSubstring(string s) {
                           if(s.size()==0||s.size()==1)return s.size();
                            int cnt=0;
                            int f_cnt=0;
                            string str="";
                            
                            
                            for(int i=0;i<s.size();i++){
                               char c=s[i];
                               size_t found=str.find(c);
                        
                                if(found!=string::npos){
                                    cnt=cnt-(found+1);
                                    str.erase(0,found+1);
                                }
                                cnt++;
                                str+=s[i];
                                f_cnt=max(cnt,f_cnt);
                            }
                            return f_cnt;
                        }
                    };

Algorithm

                class Solution {
                    public:
                        int findCharacter(string &str,char x){
                            int index=-1;
                            for(int i=0;i<str.length();i++){
                                if(str[i]==x)
                                    return i;
                            }
                            return index;
                        }
                        int lengthOfLongestSubstring(string s) {
                            int n=s.length();
                            if(n==0)
                                return 0;
                            
                            string str="";
                            int length=0;
                            for(int i=0;i<n;i++){
                                int index=findCharacter(str,s[i]);
                                if(index==-1){
                                    str=str+s[i];
                                    if(length<str.length())
                                        length=str.length();
                                }
                                else{
                                    if(length<str.length())
                                        length=str.length();
                                        str=str.erase(0,index+1)+s[i];
                                }
                            }
                            if(length==0)
                                return str.length();
                            else return length;
                        }
                    };

Problem 31

Number of Good Pairs

Algorithm

It is like u have two seats and n number of people.In how many ways they can sit on the chair. There are total nc2 ways.But one important thing one person can sit one time with the same person. 

                class Solution {
                    public:
                        int numIdenticalPairs(vector<int>& nums) {
                            int n=nums.size();
                            map<int,int> mp;
                            for(int i=0;i<n;i++){
                                mp[nums[i]]++;
                            }
                            int sum=0;
                            for(auto itr=mp.begin();itr!=mp.end();itr++){
                                if(itr->second>1){
                                    sum+=(itr->second*(itr->second-1))/2;
                                }
                            }
                            return sum;
                        }
                    };

Problem 94

Subarray Sum Equals K

Algorithm

The idea behind this approach is as follows: If the cumulative sum(represented by sum[i]sum[i] for sum up to i^{th}i thindex) up to two indices is the same, the sum of the elements lying in between those indices is zero. Extending the same thought further, if the cumulative sum up to two indices, say i and j is at a difference of k i.e. if sum[i] - sum[j] = k, the sum of elements lying between indices i and j is k.

orignal array 1 2 3
commulative sum 1 3 6. sum[2]-sum[0]=5 and if we find the addition of original array from (0,2] we get 5. Now we can also write it like sum[i]-k=sum[j]. so we just find the sum[j] in the hasmap. 

                class Solution {
                    public:
                        int subarraySum(vector<int>& nums, int k) {
                            int count=0,sum=0;
                            unordered_map<int,int> mp;
                            mp[0]=1;
                            for(int i=0;i<nums.size();i++){
                                sum=sum+nums[i];
                                 if(mp.find(sum-k)!=mp.end()){
                                     count=count+mp[sum-k];
                                 }
                                 mp[sum]++;
                                
                            }
                            return count;
                        }
                    };

Problem 95

Daily Temperatures

Algorithm

starting from first point. 

                class Solution {
                    public:
                        vector<int> dailyTemperatures(vector<int>& T) {
                            stack<pair<int,int>> s;
                            s.push(make_pair(0,T[0]));
                            for(int i=1;i<T.size();i++){
                                auto p=s.top();
                               while(!s.empty()&&p.second<T[i]){
                                   s.pop();
                                   T[p.first]=i-p.first;
                                   if(!s.empty())p=s.top();
                               }
                                s.push(make_pair(i,T[i]));
                                
                            }
                            while(!s.empty()){
                                auto p=s.top();
                                T[p.first]=0;
                                s.pop();
                            }
                            return T;
                        }
                    };

Algorithm

starting from ending point. 

                class Solution {
                    public:
                        vector<int> dailyTemperatures(vector<int>& T) {
                            stack<pair<int,int>> s;
                            vector<int> v(T.size());
                    
                            for(int i=T.size()-1;i>=0;i--){
                             
                                while(!s.empty()&&(s.top().first<=T[i]))
                                    s.pop();
                                  
                                v[i]=(s.empty()?0:(s.top().second-i));    
                                s.push(make_pair(T[i],i));
                            }
                            return v;
                        }
                    };

                            if(root==NULL)return true;

Problem 97

Maximum Width of Binary Tree

Algorithm

specific level completely popped at single time. 

                class Solution {
                    public:
                        int widthOfBinaryTree(TreeNode* root) {
                            queue<pair<TreeNode*,unsigned int>> s;
                            int max_val=-1;
                            s.push(make_pair(root,0));
                            while(!s.empty()){
                                int size=s.size();
                                int k=size;
                                 int first=0,last=0;
                                while(size--){
                                    auto p=s.front();
                                    s.pop();
                                    if(p.first->left)s.push(make_pair(p.first->left,2*p.second+1));
                                    if(p.first->right)s.push(make_pair(p.first->right,2*p.second+2));
                                    
                                   
                                    if(size==k-1){
                                        first=p.second;
                                    }
                                    if(size==0)last=p.second;
                                    
                                 
                                    
                                }
                                   max_val=max(max_val,last-first+1);
                                
                            }
                            return max_val;
                        }
                    };

                class Solution {
                    public:
                        int widthOfBinaryTree(TreeNode* root) {
                            queue<pair<TreeNode*,unsigned int>> q;
                            q.push(make_pair(root,0));
                            TreeNode* dummy=new TreeNode(-1);
                            q.push(make_pair(dummy,-1));
                            unsigned int dist=LONG_MIN;
                            unsigned int s=LONG_MAX;
                            unsigned int m=LONG_MIN;
                            while(!q.empty()){
                                auto p=q.front();
                                q.pop();
                                 s=p.second<s?p.second:s;
                                 m=p.second>m?p.second:m;
                                auto d=q.front();
                                
                                if(p.first->left)
                                    q.push(make_pair(p.first->left,2*p.second+1));
                                if(p.first->right)
                                    q.push(make_pair(p.first->right,2*p.second+2));
                                if(d.second==-1){
                                    dist=dist>m-s+1?dist:m-s+1;
                                    s=LONG_MAX;
                                    m=LONG_MIN;
                                    q.pop();
                                    q.push(make_pair(dummy,-1));
                                }
                            }
                            return dist;
                        }
                    };

Path Sum II

Algorithm

  
class Solution {
    public:
        vector<vector<int>> v;
        vector<int> rv;
        vector<vector<int>> pathSum(TreeNode* root, int targetSum) {
            if(root==NULL)return {};
            if(root->left==NULL&&root->right==NULL){
                if(root->val==targetSum){
                    rv.push_back(targetSum);
                    v.push_back(rv);
                    rv.pop_back();
                }
                  
                
                return v;
            }
            rv.push_back(root->val);
            pathSum(root->left,targetSum-root->val);
            pathSum(root->right,targetSum-root->val);
            rv.pop_back();
    
           
            
            return v;
        }
    };

                                    if(target-root->val==0){

                            bool left=check(root->left,target-root->val,v,rv);
                            
                            bool right=check(root->right,target-root->val,v,rv);
                           
                            
                            return left||right;
                            
                        }
                        vector<vector<int>> pathSum(TreeNode* root, int targetSum) {

                            vector<int> rv;
                            
                            if(check(root,targetSum,v,rv))
                                return v;
                            
                            return {};
                        }
                    };
